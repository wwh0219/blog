// reactiveGetter代码片段
Object.defineProperty(obj, key, {
  enumerable: true,
  configurable: true,
  get: function reactiveGetter () {
    const value = getter ? getter.call(obj) : val
    if (Dep.target) {
      dep.depend()
      if (childOb) {
        childOb.dep.depend()
        if (Array.isArray(value)) {
          dependArray(value)
        }
      }
    }
    return value
  },

分析上文的代码，首次执行reactiveGetter之前调用了pushTarget将Dep.target设为了当前实例的渲染Watcher，此时执行dep.depend()，如下

// Dep#depend
depend () {
    if (Dep.target) {
      Dep.target.addDep(this)
    }
}
// Watcher#addDep
addDep (dep: Dep) {
  const id = dep.id
  if (!this.newDepIds.has(id)) {
    this.newDepIds.add(id)
    this.newDeps.push(dep)
    if (!this.depIds.has(id)) {
      dep.addSub(this)
    }
  }
}
